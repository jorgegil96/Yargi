class Carro() {

    int a, b;

    fun factorial(int n, int m): int {
        int x;
        if (n <= 1) {
            x = 1;
        } else {
            x = n * factorial(n - 1, m);
        }
        return x;
    }

    fun sum(int p, int q): int {
        return p + q;
    }

	main() {
	    a = factorial(4 + 1, 1 + 1);
	}
}
